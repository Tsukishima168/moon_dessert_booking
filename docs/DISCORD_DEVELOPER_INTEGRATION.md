完整的訂單 POST 路由\n\n```typescript\nimport { NextRequest, NextResponse } from 'next/server';\nimport { createAuthClient } from '@/lib/supabase/server-auth';\nimport { notifyOrderToDiscord } from '@/lib/discord-notifications';\n\nexport async function POST(req: NextRequest) {\n    const supabase = await createAuthClient();\n    \n    try {\n        const data = await req.json();\n        \n        // 驗證數據\n        if (!data.customer_name || !data.items) {\n            return NextResponse.json(\n                { error: '缺少必要字段' },\n                { status: 400 }\n            );\n        }\n        \n        // 保存訂單\n        const { data: order, error } = await supabase\n            .from('orders')\n            .insert({\n                customer_name: data.customer_name,\n                phone: data.phone,\n                email: data.email,\n                items: data.items,\n                total_price: data.total_price,\n                status: 'pending',\n                created_at: new Date().toISOString(),\n            })\n            .select()\n            .single();\n        \n        if (error) {\n            return NextResponse.json(\n                { error: '保存訂單失敗' },\n                { status: 500 }\n            );\n        }\n        \n        // 發送 Discord 通知（不應阻止主流程）\n        try {\n            await notifyOrderToDiscord(order, 'created');\n        } catch (discordError) {\n            console.warn('Discord 通知失敗，但訂單已保存:', discordError);\n        }\n        \n        // 返回成功\n        return NextResponse.json({\n            success: true,\n            order,\n            message: '訂單已建立',\n        });\n        \n    } catch (error) {\n        console.error('未預期的錯誤:', error);\n        return NextResponse.json(\n            { error: '伺服器錯誤' },\n            { status: 500 }\n        );\n    }\n}\n```\n\n---\n\n## 🧪 測試 Discord 通知\n\n### 手動測試\n\n```bash\n# 1. 訪問後台 Discord 設定頁面\nhttps://your-app.com/admin/discord-settings\n\n# 2. 輸入 Webhook
